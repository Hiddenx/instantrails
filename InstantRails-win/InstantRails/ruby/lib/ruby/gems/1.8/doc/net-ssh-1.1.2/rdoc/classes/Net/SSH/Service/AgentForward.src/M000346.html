        def register_services( container )

          container.namespace_define :agentforward do |ns|
            ns.driver :model => :singleton_deferred do |c,p|
              require 'net/ssh/service/agentforward/driver'
              Driver.new( c[:connection][:driver],
                          c[:transport][:buffers],
                          c[:log_for, p],
                          c[:userauth].agent)
            end
          end

        end
