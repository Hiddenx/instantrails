          def get_lengths( name )
            ossl_name = @cipher_map[ name ]
            return [ 0, 0 ] if ossl_name.nil? || ossl_name == "none"

            cipher = OpenSSL::Cipher::Cipher.new( ossl_name )
            return [ cipher.key_len, cipher.block_size ]
          end
