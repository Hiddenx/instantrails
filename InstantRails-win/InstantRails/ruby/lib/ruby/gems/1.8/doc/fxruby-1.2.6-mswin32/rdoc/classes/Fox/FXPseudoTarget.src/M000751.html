    def onHandleMsg(sender, sel, ptr)
      messageType = Fox.FXSELTYPE(sel)
      result = @blocks[messageType].call(sender, sel, ptr)
      case messageType
        when SEL_TIMEOUT
          @@targets_of_pending_timers.delete(self)
        when SEL_CHORE
          @@targets_of_pending_chores.delete(self)
      end
      result
    end
