        def restart
                setFocus
                setText("")
                # IRB initialization
                @inputAdded = 0
                @input = IO.pipe
                $DEFAULT_OUTPUT = self

                @im = FXIRBInputMethod.new
                if @irb
                        @irb.kill
                        @irb = Thread.new {
                                IRB.restart(@im)
                        }
                else
                        @irb = Thread.new {
                                IRB.start_in_fxirb(@im)
                        }
                end
        end
