        def redir
                @output = IO.pipe
                @old_out = $defout.dup
                @old_err = $stderr.dup
                @old_in = $stdin.dup
                $stdin = @dest.input[0]
                $defout = $stderr = @output[1]
                @output[1].fcntl(Fcntl::F_SETFL, File::NONBLOCK)
                @thread = Thread.new {
                        while not @output[0].eof?
                                select([@output[0]])
                                @dest.write(@output[0].read)
                        end
                        @output[0].close
                }
        end
