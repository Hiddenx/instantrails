    def capture(&block)
      assigns = instance_variables.inject({}) do |hsh, iv|
        unless ['@stream', '@builder', '@assigns', '@helpers'].include?(iv)
          hsh[iv[1..-1]] = instance_variable_get(iv)
        end
        hsh
      end
      self.class.new(assigns, @helpers, &block).to_s
    end
