    def info
      div :id=>"viewport" do
        @services.each do |s|
        
          h2 { "#{s.service_name} service information" }
          table do
            tr { th {"Attribute"}; th {"Value"} }
            
            s.each_pair do |memb,obj|
              name = memb.to_s.tr("_"," ").capitalize
              tr { 
                td { b { "#{name}: " } }
                td { obj.inspect }
              }
            end
          end
        end
      end
    end
