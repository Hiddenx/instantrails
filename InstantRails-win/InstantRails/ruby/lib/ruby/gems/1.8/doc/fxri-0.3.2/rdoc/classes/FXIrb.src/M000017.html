        def processCommandLine(cmd)
                @multiline = false
                lines = cmd.split(/\n/)
                lines.each {|i| 
                        @input[1].puts i
                        @inputAdded += 1
                }
                @im.print_prompt = false
                while (@inputAdded > 0) do
                        @irb.run
                end
                @im.merge_last(lines.length)
                @im.print_prompt = true 
        end
