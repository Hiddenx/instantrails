  def stop
    if _running?
      ul do
        li { a "Stop (TERM)", :href => "/kill/term" }
        li { a "Reload (HUP)", :href => "/kill/hup" }
        li { a "Restart (USR2)", :href => "/kill/usr2" }
        li { a "Kill (KILL)", :href => "/kill/kill" }
      end
    else
      p { "Mongrel does not appear to be running (no PID file at #$PID_FILE)." }
    end
  end
