  def logs
    div do
      h2 { "Logs" }
      table do
        tr do
          th { "File"}; th { "Bytes" }; th { "Last Modified" }
        end
        @log_files.each do |file|
          tr do
            td { a file, :href => "../#{file}" }
            td { File.size file }
            td { File.mtime file }
          end
        end
      end
    end
  end
