          def initialize( connection, log, pty_opts )
            @connection = connection
            @log = log

            @pty_opts = pty_opts

            @stdout = ""
            @stderr = ""

            @state = :opening
            @connection.open_channel( "session", &method( :on_confirm ) )

            @connection.loop { @state != :open && @state != :closed }
            raise "could not open shell" if @state != :open
          end
