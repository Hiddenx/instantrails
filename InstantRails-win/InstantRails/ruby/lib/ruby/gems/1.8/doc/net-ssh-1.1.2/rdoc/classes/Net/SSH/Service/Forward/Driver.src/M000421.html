          def cancel_local( local_port, bind_address="127.0.0.1" )
            key = [ local_port.to_i, bind_address ]

            forward = @local_forwards[ key ]
            @local_forwards.delete key

            forward[ :socket ].shutdown
            forward[ :thread ].terminate

            true
          end
