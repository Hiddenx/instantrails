        def collect_thread(thread)
            begin
                thread.join
            rescue Interrupt
                @log.info("Shutting down from SIGINT.")
            rescue IOError
                @log.error("received IOError #$!.  Web server may possibly be configured wrong.")
            rescue Object
                @log.error("Collecting thread", $!)
            end
        end
