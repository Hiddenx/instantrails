        def configure
            salting = ('a' .. 'z').to_a + ('A' .. 'Z').to_a + ('0' .. '9').to_a
            @settings[:password] = @settings[:password].crypt(salting[rand(salting.length)] + salting[rand(salting.length)])
            open(@settings[:config],"w") {|f| f.write(YAML.dump(@settings)) }
        end
