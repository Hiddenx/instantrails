          def generate_key_fingerprint(key)
            writer = @buffers.writer
            writer.write_key(key)

            blob = writer.to_s
            fingerprint = OpenSSL::Digest::MD5.hexdigest(blob).scan(/../).join(":")

            [blob, fingerprint]
          rescue ::Exception => e
            [nil, "(could not generate fingerprint: #{e.message})"]
          end
