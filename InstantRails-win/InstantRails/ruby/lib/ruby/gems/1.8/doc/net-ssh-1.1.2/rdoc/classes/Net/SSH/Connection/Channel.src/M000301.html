        def send_request( request_name, data, want_reply=false )
          msg = @buffers.writer
          msg.write_byte CHANNEL_REQUEST
          msg.write_long @remote_id
          msg.write_string request_name
          msg.write_bool want_reply
          msg.write data.to_s
          @connection.send_message msg
          self
        end
