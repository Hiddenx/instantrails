514:     def process_client(client)
515:       begin
516:         parser = HttpParser.new
517:         params = {}
518:         request = nil
519:         data = client.readpartial(Const::CHUNK_SIZE)
520:         nparsed = 0
521: 
522:         # Assumption: nparsed will always be less since data will get filled with more
523:         # after each parsing.  If it doesn't get more then there was a problem
524:         # with the read operation on the client socket.  Effect is to stop processing when the
525:         # socket can't fill the buffer for further parsing.
526:         while nparsed < data.length
527:           nparsed = parser.execute(params, data, nparsed)
528: 
529:           if parser.finished?
530:             script_name, path_info, handlers = @classifier.resolve(params[Const::REQUEST_URI])
531: 
532:             if handlers
533:               params[Const::PATH_INFO] = path_info
534:               params[Const::SCRIPT_NAME] = script_name
535:               params[Const::REMOTE_ADDR] = params[Const::HTTP_X_FORWARDED_FOR] || client.peeraddr.last
536:               notifier = handlers[0].request_notify ? handlers[0] : nil
537: 
538:               # TODO: Find a faster/better way to carve out the range, preferably without copying.
539:               data = data[nparsed ... data.length] || ""
540: 
541:               request = HttpRequest.new(params, data, client, notifier)
542: 
543:               # in the case of large file uploads the user could close the socket, so skip those requests
544:               break if request.body == nil  # nil signals from HttpRequest::initialize that the request was aborted
545: 
546:               # request is good so far, continue processing the response
547:               response = HttpResponse.new(client)
548: 
549:               # Process each handler in registered order until we run out or one finalizes the response.
550:               handlers.each do |handler|
551:                 handler.process(request, response)
552:                 break if response.done or client.closed?
553:               end
554: 
555:               # And finally, if nobody closed the response off, we finalize it.
556:               unless response.done or client.closed? 
557:                 response.finished
558:               end
559:             else
560:               # Didn't find it, return a stock 404 response.
561:               client.write(Const::ERROR_404_RESPONSE)
562:             end
563: 
564:             break #done
565:           else
566:             # Parser is not done, queue up more data to read and continue parsing
567:             data << client.readpartial(Const::CHUNK_SIZE)
568:             if data.length >= Const::MAX_HEADER
569:               raise HttpParserError.new("HEADER is longer than allowed, aborting client early.")
570:             end
571:           end
572:         end
573:       rescue EOFError,Errno::ECONNRESET,Errno::EPIPE,Errno::EINVAL,Errno::EBADF
574:         # ignored
575:       rescue HttpParserError
576:         STDERR.puts "#{Time.now}: BAD CLIENT (#{params[Const::HTTP_X_FORWARDED_FOR] || client.peeraddr.last}): #$!"
577:       rescue Errno::EMFILE
578:         reap_dead_workers('too many files')
579:       rescue Object
580:         STDERR.puts "#{Time.now}: ERROR: #$!"
581:       ensure
582:         client.close unless client.closed?
583:         request.body.delete if request and request.body.class == Tempfile
584:       end
585:     end
