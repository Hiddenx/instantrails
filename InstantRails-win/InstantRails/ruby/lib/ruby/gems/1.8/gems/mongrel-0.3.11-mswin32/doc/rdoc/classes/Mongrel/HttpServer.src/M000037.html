418:     def run
419:       BasicSocket.do_not_reverse_lookup=true
420:       @acceptor = Thread.new do
421:         Thread.current[:stopped] = false
422: 
423:         while not Thread.current[:stopped]
424:           begin
425:             @req_queue << @socket.accept
426:           rescue StopServer
427:             STDERR.puts "Server stopped.  Exiting."
428:             @socket.close if not @socket.closed?
429:             break
430:           rescue Errno::EMFILE

432:             sleep 0.5
433:           end
434:         end
435: 
436:         # now that processing is done we feed enough false onto the request queue to get
437:         # each processor to exit and stop processing.
438:         @processors.length.times { @req_queue << false }
439: 
440:         # finally we wait until the queue is empty
441:         while @req_queue.length > 0
442:           STDERR.puts "Shutdown waiting for #{@req_queue.length} requests" if @req_queue.length > 0
443:           sleep 1
444:         end
445:       end
446: 
447:       @acceptor.priority = 1
448: 
449:       return @acceptor
