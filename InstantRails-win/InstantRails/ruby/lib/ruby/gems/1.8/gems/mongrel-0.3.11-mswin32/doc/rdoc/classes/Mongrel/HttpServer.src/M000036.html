373:     def process_client(client)
374:       begin
375:         parser = HttpParser.new
376:         params = {}
377:         data = client.readpartial(Const::CHUNK_SIZE)
378: 
379:         while true
380:           nread = parser.execute(params, data)
381:           if parser.finished?
382:             script_name, path_info, handler = @classifier.resolve(params[Const::REQUEST_URI])
383: 
384:             if handler
385:               params[Const::PATH_INFO] = path_info
386:               params[Const::SCRIPT_NAME] = script_name
387:               request = HttpRequest.new(params, data[nread ... data.length], client)
388:               response = HttpResponse.new(client)
389:               handler.process(request, response)
390:             else
391:               client.write(Const::ERROR_404_RESPONSE)
392:             end
393:             
394:             break #done
395:           else
396:             # gotta stream and read again until we can get the parser to be character safe
397:             # TODO: make this more efficient since this means we're parsing a lot repeatedly
398:             parser.reset
399:             data << client.readpartial(Const::CHUNK_SIZE)
400:           end
401:         end
402:       rescue EOFError
403:         # ignored
404:       rescue Errno::ECONNRESET
405:         # ignored
406:       rescue Errno::EPIPE
407:         # ignored
408:       rescue => details
409:         STDERR.puts "ERROR(#{details.class}): #{details}"
410:         STDERR.puts details.backtrace.join("\n")
411:       ensure
412:         client.close
413:       end
414:     end
