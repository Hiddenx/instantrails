170:       def run(args)
171:         # find the command
172:         cmd_name = args.shift
173: 
174:         if !cmd_name or cmd_name == "?" or cmd_name == "help"
175:           print_command_list
176:           return true
177:         elsif cmd_name == "--version"
178:           STDERR.puts "Mongrel Web Server #{Mongrel::Const::MONGREL_VERSION}"
179:           return true
180:         end
181: 
182:         # command exists, set it up and validate it
183:         begin
184:           command = GemPlugin::Manager.instance.create("/commands/#{cmd_name}", :argv => args)
185:         rescue
186:           STDERR.puts "INVALID COMMAND: #$!"
187:           print_command_list
188:           return false
189:         end
190: 
191:         # Normally the command is NOT valid right after being created
192:         # but sometimes (like with -h or -v) there's no further processing
193:         # needed so the command is already valid so we can skip it.
194:         if not command.done_validating
195:           if not command.validate
196:             STDERR.puts "#{cmd_name} reported an error. Use mongrel_rails #{cmd_name} -h to get help."
197:             return false
198:           else
199:             command.run
200:           end
201:         end
202: 
203:         return true
204:       end
