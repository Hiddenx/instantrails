137:       def run(args)
138:         # find the command
139:         cmd_name = args.shift
140:         
141:         if !cmd_name or cmd_name == "?" or cmd_name == "help"
142:           print_command_list
143:           return true
144:         end
145:         
146:         # command exists, set it up and validate it
147:         begin
148:           command = GemPlugin::Manager.instance.create("/commands/#{cmd_name}", :argv => args)
149:         rescue
150:           STDERR.puts "INVALID COMMAND: #$!"
151:           print_command_list
152:           return
153:         end
154:         
155:         # Normally the command is NOT valid right after being created
156:         # but sometimes (like with -h or -v) there's no further processing
157:         # needed so the command is already valid so we can skip it.
158:         if not command.done_validating
159:           if not command.validate
160:             STDERR.puts "#{cmd_name} reported an error. Use -h to get help."
161:             return false
162:           else
163:             command.run
164:           end
165:         end
166:         return true
167:       end
