232:         def handle_client(socket)
233:             # ruby's GC seems to do weird things if we don't assign the thread to a local variable
234:             @threads << Thread.new do
235:                 begin
236:                     len = ""
237:                     # we only read 10 bytes of the length.  any request longer than this is invalid
238:                     while len.length <= 10
239:                         c = socket.read(1)
240:                         if c == ':'
241:                             # found the terminal, len now has a length in it so read the payload
242:                             break
243:                         else
244:                             len << c
245:                         end
246:                     end
247:                     
248:                     # we should now either have a payload length to get
249:                     payload = socket.read(len.to_i)
250:                     if (c = socket.read(1)) != ','
251:                         @log.error("Malformed request, does not end with ','")
252:                     else
253:                         read_header(socket, payload)
254:                     end
255:                 rescue Object
256:                     @log.error("Handling client", $!)
257:                 ensure
258:                     # no matter what we have to put this thread on the bad list
259:                     socket.close if not socket.closed?
260:                 end
261:             end
262:         end
