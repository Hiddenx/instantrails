248:     def process(request, response)
249:       req_method = request.params[Const::REQUEST_METHOD] || Const::GET
250:       req_path = can_serve request.params[Const::PATH_INFO]
251:       if not req_path
252:         # not found, return a 404
253:         response.start(404) do |head,out|
254:           out << "File not found"
255:         end
256:       else
257:         begin
258:           if File.directory? req_path
259:             send_dir_listing(request.params[Const::REQUEST_URI], req_path, response)
260:           elsif req_method == Const::HEAD
261:             send_file(req_path, request, response, true)
262:           elsif req_method == Const::GET
263:             send_file(req_path, request, response, false)
264:           else
265:             response.start(403) {|head,out| out.write(ONLY_HEAD_GET) }
266:           end
267:         rescue => details
268:           STDERR.puts "Error sending file #{req_path}: #{details}"
269:         end
270:       end
271:     end
